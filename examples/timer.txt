### Set up interrupt vectors
loadwi 15 0

# first vector
strapr 15 .intr_vec

# Point to next vector
addi 15 1 15

# Remaining vectors
.write_vector
strapr 15 .exit
addi 15 1 15
jmpgei 15 10 .write_done  # 16 vectors
jmp .write_vector  # Next vector

.write_done

eni  # Reenable interrupts


### Set up interrupt controller
savewi 20 FFFFFED2
savewi .timer_intr FFFFFED6
savebi 1 FFFFFEDA
savewi 0 FFFFFECE

### Set up timer
savewi 64 FFFFFEC9

### Main loop
.loop
nop
jmplti A F .loop
halt


### Timer interrupt
.timer_intr
addi A 1 A
savewi 0 FFFFFECE  # Re-enable interrupts
ret

### Interrupt vectors
.intr_vec
jmp FFFFFEEA
